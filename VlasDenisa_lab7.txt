//1. Se citesc trei numere de la tastatură (a, b şi c). Să se determine aria dreptunghiului ale cărui laturi sunt a şi b şi verificaţi dacă diagonala dreptunghiului este egală cu c.

#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
using namespace std;

int arieDr(int a, int b);
int diagonalaDr(int a, int b);
int main()
{
	int a, b, c,arie,diag;
	printf("\n primul numar este= ");
	scanf("%d", &a);
	printf("\n al doilea numar este= ");
	scanf("%d", &b);
	printf("\n al treilea numar este= ");
	scanf("%d", &c);
	arie = arieDr(a, b);
	diag = diagonalaDr(a, b);
	printf("\n aria dreptunghiului este egala cu= %d", arie);
	if (diag == c)
		printf("\n diagonala dreptunghiului este egala cu c");
	else
		printf("\n diagonala dreptunghiului nu este egala cu c");
}
int arieDr(int a, int b)
{
	return a * b;
}
int diagonalaDr(int a, int b)
{
	return sqrt(a * a + b * b);
}

//Vlas Denisa 2112, ex 4 lab 7
//4. Să se scrie un program care citeşte de la tastatură o valoare întreagă n şi calculează n! (n-factorial).
#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
using namespace std;
int factorial(int n);
int main()
{
	int n, val;
	printf("\n n= ");
	scanf("%d", &n);
	val = factorial(n);
	printf("\n n factorial este= %d", val);
}
int factorial(int n)
{
	int rez = 1;
	for (int i = n; i >= 1; i--)
		rez = rez * i;
	return rez;
}
